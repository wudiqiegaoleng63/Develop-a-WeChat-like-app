package model

type GroupInfo struct {

    CreatedAt time.Time       `gorm:"column:created_at;index;type:datetime;not null;comment:创建时间"`

```go
// internal/service/gorm/group_info_service.go
package gorm

import (
	"encoding/json"
	"errors"
	"fmt"
	"time"

	"github.com/go-redis/redis/v8"
	"gorm.io/gorm"

	"kama-chat-server/internal/dao"
	"kama-chat-server/internal/dto/request"
	"kama-chat-server/internal/dto/respond"
	"kama-chat-server/internal/model"
	myredis "kama-chat-server/internal/service/redis"
	"kama-chat-server/pkg/constants"
	"kama-chat-server/pkg/enum/contact/contact_status_enum"
	"kama-chat-server/pkg/enum/contact/contact_type_enum"
	"kama-chat-server/pkg/enum/group_info/group_status_enum"
	"kama-chat-server/pkg/util/random"
	"kama-chat-server/pkg/zlog"
)

type groupInfoService struct{}

var GroupInfoService = new(groupInfoService)

// CreateGroup 创建群聊
// ★ 重点： Members字段的JSON序列化处理
func (g *groupInfoService) CreateGroup(groupReq request.CreateGroupRequest) (string, int) {
	// 1. 构建GroupInfo对象
	group := model.GroupInfo{
		// ★ 群组Uuid格式: "G" + 11位随机字符
		Uuid:      fmt.Sprintf("G%s", random.GetNowAndLenRandomString(11)),
		Name:      groupReq.Name,
		Notice:    groupReq.Notice,
		OwnerId:   groupReq.OwnerId,
		MemberCnt: 1,                          // 初始群人数为1（只有群主）
		AddMode:   groupReq.AddMode,
		Avatar:    groupReq.Avatar,
		Status:    group_status_enum.NORMAL,   // 状态: 正常
		CreatedAt: time.Now(),
		UpdatedAt: time.Now(),
	}

	// 2. ★★★ Members字段的JSON序列化 ★★★
	// 初始成员数组只有群主一人
	var members []string
	members = append(members, groupReq.OwnerId)
	// 使用json.Marshal将[]string序列化为[]byte(json.RawMessage)
	var err error
	group.Members, err = json.Marshal(members)
	if err != nil {
		zlog.Error(err.Error())
		return constants.SYSTEM_ERROR, -1
	}

	// 3. 创建群聊记录
	if res := dao.GormDB.Create(&group); res.Error != nil {
		zlog.Error(res.Error.Error())
		return constants.SYSTEM_ERROR, -1
	}

	// 4. 为群主添加联系人记录
	// 群主与群聊之间也需要建立UserContact关系
	contact := model.UserContact{
		UserId:      groupReq.OwnerId,              // 群主的uuid
		ContactId:   group.Uuid,                    // 群聊的uuid
		ContactType: contact_type_enum.GROUP,       // 联系类型: 群聊
		Status:      contact_status_enum.NORMAL,    // 状态: 正常
		CreatedAt:   time.Now(),
		UpdateAt:    time.Now(),
	}
	if res := dao.GormDB.Create(&contact); res.Error != nil {
		zlog.Error(res.Error.Error())
		return constants.SYSTEM_ERROR, -1
	}

	// ★5. 删除Redis缓存，下次查询会重新从数据库读取最新数据
	if err := myredis.DelKeysWithPattern("contact_mygroup_list_" + groupReq.OwnerId); err != nil {
		zlog.Error(err.Error())
	}

	return "创建成功", 0
}
```
### 3.6 数据库操作说明
| 操作 | 数据表 | 说明 |
|------|------|------|
| Create | group_info | 插入新群聊记录 |
| Create | user_contact | 为群主创建与群聊的联系人记录 |
### 3.7 关键概念详解
#### ★★★ Members的json.Marshal序列化 ★★★
GORM框架不支持直接将`[]string`类型存储到数据库中，因此需要使用`json.Marshal`进行序列化转换。
```go
// 序列化过程
var members []string = ["U12345678901"]
membersBytes, err := json.Marshal(members)
// 结果: ["U12345678901"] 存入数据库

// 反序列化过程（读取时）
var members []string
err = json.Unmarshal(group.Members, &members)
// 结果: members = ["U12345678901"]
```
**为什么要序列化?**
- GORM不支持复杂类型（如数组、切片）的直接存储
- `json.RawMessage`底层是`[]byte`类型，可以直接存储JSON字符串
- 序列化保证了数据的一致性和可解析性
#### ★ 群组Uuid格式
```go
Uuid: fmt.Sprintf("G%s", random.GetNowAndLenRandomString(11))
// 例如: G20260420123 (G + 11位时间戳随机字符串)
```
- 以"G"开头标识这是一个群组
- 11位随机字符保证唯一性
- 与用户Uuid（以"U"开头）区分开来
#### ★ 初始Members只有群主
```go
// 创建群聊时，Members数组中只有群主一人
var members []string
members = append(members, groupReq.OwnerId)  // 只有群主的uuid
```
**设计原因:**
- 群聊刚创建时，只有创建者（群主）一人
- 其他成员通过后续的"加入群聊"操作添加
- 保证了数据的初始一致性

### 4.5 Service层完整代码
```go
// internal/service/gorm/group_info_service.go

// LoadMyGroup 获取我创建的群聊
// ★返回群聊列表DTO，带Redis缓存
func (g *groupInfoService) LoadMyGroup(ownerId string) (string, []respond.LoadMyGroupRespond, int) {
	// ★1. 先从Redis缓存读取
	rspString, err := myredis.GetKeyNilIsErr("contact_mygroup_list_" + ownerId)
	if err != nil {
		// 缓存不存在，从数据库查询
		if errors.Is(err, redis.Nil) {
			// 2. 查询该用户创建的所有群聊
			var groupList []model.GroupInfo
			// 按创建时间倒序排列（最新的在前）
			if res := dao.GormDB.Order("created_at DESC").Where("owner_id = ?", ownerId).Find(&groupList); res.Error != nil {
				zlog.Error(res.Error.Error())
				return constants.SYSTEM_ERROR, nil, -1
			}

			// 3. 构建响应DTO列表
			var groupListRsp []respond.LoadMyGroupRespond
			for _, group := range groupList {
				groupListRsp = append(groupListRsp, respond.LoadMyGroupRespond{
					GroupId:   group.Uuid,
					GroupName: group.Name,
					Avatar:    group.Avatar,
				})
			}

			// ★4. 序列化并写入缓存
			rspString, err := json.Marshal(groupListRsp)
			if err != nil {
				zlog.Error(err.Error())
			}
			if err := myredis.SetKeyEx("contact_mygroup_list_"+ownerId, string(rspString), time.Minute*constants.REDIS_TIMEOUT); err != nil {
				zlog.Error(err.Error())
			}

			return "获取成功", groupListRsp, 0
		} else {
			zlog.Error(err.Error())
			return constants.SYSTEM_ERROR, nil, -1
		}
	}

	// ★5. 缓存存在，直接反序列化返回
	var groupListRsp []respond.LoadMyGroupRespond
	if err := json.Unmarshal([]byte(rspString), &groupListRsp); err != nil {
		zlog.Error(err.Error())
	}
	return "获取成功", groupListRsp, 0
}
```
### 4.6 数据库操作说明
| 操作 | 数据表 | 查询条件 | 说明 |
|------|------|------|------|
| Find | group_info | owner_id = ? | 查询指定用户创建的群聊 |
| Order | group_info | created_at DESC | 按创建时间倒序排列 |
### 4.7 关键概念详解
#### ★ 查询条件
```go
Where("owner_id = ?", ownerId)
```
- 通过`owner_id`字段筛选出该用户创建的群聊
- 只查询群主是自己创建的群聊，不包括加入的群聊
#### ★ 排序方式
```go
Order("created_at DESC")
```
- 按创建时间倒序排列
- 最新创建的群聊显示在列表最前面
- 提升用户体验，最近操作的群聊更容易找到
#### ★ DTO转换
```go
// 从DAO转换为DTO，只返回必要字段
for _, group := range groupList {
    groupListRsp = append(groupListRsp, respond.LoadMyGroupRespond{
        GroupId:   group.Uuid,      // 群聊ID

        Avatar:    group.Avatar,   // 群聊头像
    })
}
```
- 不返回所有字段，只返回前端需要的展示信息
- 减少数据传输量，提高性能

// internal/service/gorm/group_info_service.go

// CheckGroupAddMode 检查群聊加群方式
// ★返回AddMode数值: 0=直接加入, 1=需要审核，带Redis缓存
func (g *groupInfoService) CheckGroupAddMode(groupId string) (string, int8, int) {
	// ★1. 先从Redis缓存读取群聊详情
	rspString, err := myredis.GetKeyNilIsErr("group_info_" + groupId)
	if err != nil {
		// 缓存不存在，从数据库查询
		if errors.Is(err, redis.Nil) {
			// 2. 查询群聊信息
			var group model.GroupInfo
			if res := dao.GormDB.First(&group, "uuid = ?", groupId); res.Error != nil {
				zlog.Error(res.Error.Error())
				return constants.SYSTEM_ERROR, -1, -1
			}

			// 3. 返回加群方式
			return "加群方式获取成功", group.AddMode, 0
		} else {
			zlog.Error(err.Error())
			return constants.SYSTEM_ERROR, -1, -1
		}
	}

	// ★4. 缓存存在，从缓存数据中解析AddMode
	var rsp respond.GetGroupInfoRespond
	if err := json.Unmarshal([]byte(rspString), &rsp); err != nil {
		zlog.Error(err.Error())
	}
	return "加群方式获取成功", rsp.AddMode, 0
}
```

### 5.7 关键概念详解
#### ★ AddMode枚举值说明
| 值 | 含义 | 说明 |
|------|------|------|
| 0 | 直接加入 | 用户可直接入群，无需审核 |
| 1 | 需要审核 | 用户申请入群后，需要等待群主审核 |
#### ★ 业务流程影响
``go
// 如果addMode == 0 (直接加入)
// -> 调用EnterGroupDirectly接口，直接入群
// -> 不创建ContactApply记录

// 如果addMode == 1 (需要审核)
// -> 创建ContactApply记录，等待群主审核
// -> 群主在管理组件中审核申请
```
#### ★ 为什么单独设计这个接口?
1. **分离关注点**: 加群方式查询是独立操作，不与入群操作耦合
2. **前端决策**: 前端可以根据结果决定显示不同的申请界面
3. **权限控制**: 不同加群方式对应不同的权限验证逻辑

### 6.7 关键概念详解
#### ★ DeletedAt软删除判断
```go
// DeletedAt是gorm.DeletedAt类型
type DeletedAt struct {
    Time  time.Time
    Valid bool  // Valid为true表示已软删除
}
// 判断逻辑
if group.DeletedAt.Valid {
    rsp.IsDeleted = true   // 群聊已解散

    rsp.IsDeleted = false  // 群聊正常
}
```
**为什么要返回IsDeleted?**
- 前端可能需要展示"群聊已解散"的状态
- 用户可能查看已解散群聊的历史记录
- 区分正常群聊和已解散群聊的展示方式
#### ★ 不返回Members的原因
```go
// 响应中不包含Members字段

// 2. 获取成员列表有专门的接口(GetGroupMemberList)
// 3. 分离关注点，提高性能
```
#### ★ 响应字段设计原则
- 只返回前端展示需要的字段
- 不返回敏感或大数据字段
- 保持接口简洁高效

## 八、Redis缓存机制（重要）

### 8.1 缓存策略概述

群组相关接口使用Redis缓存提高查询性能，减少数据库压力。

**缓存原则**：
- 读操作：先查缓存，缓存不存在再查数据库，然后写入缓存
- 写操作：修改数据后删除相关缓存，保证数据一致性

### 8.2 缓存键格式

| 缓存键 | 用途 | 示例 |
|--------|------|------|
| `contact_mygroup_list_` + ownerId | 用户创建的群聊列表 | `contact_mygroup_list_U12345678901` |
| `group_info_` + groupId | 群聊详情信息 | `group_info_G12345678901` |
| `group_memberlist_` + groupId | 群聊成员列表 | `group_memberlist_G12345678901` |
| `group_session_list_` + userId | 用户群聊会话列表 | `group_session_list_U12345678901` |
| `my_joined_group_list_` + userId | 用户加入的群聊列表 | `my_joined_group_list_U12345678901` |

### 8.3 缓存过期时间

```go
// pkg/constants/constants.go
const REDIS_TIMEOUT = 1  // Redis超时时间（分钟）

// 使用示例
myredis.SetKeyEx("contact_mygroup_list_"+ownerId, string(rspString), time.Minute*constants.REDIS_TIMEOUT)
```

### 8.4 缓存读取流程

```go
// LoadMyGroup 获取我创建的群聊 - 带缓存
func (g *groupInfoService) LoadMyGroup(ownerId string) (string, []respond.LoadMyGroupRespond, int) {
    // 1. 先尝试从Redis读取
    rspString, err := myredis.GetKeyNilIsErr("contact_mygroup_list_" + ownerId)
    if err != nil {
        // 2. 缓存不存在，从数据库查询
        if errors.Is(err, redis.Nil) {
            var groupList []model.GroupInfo
            dao.GormDB.Order("created_at DESC").Where("owner_id = ?", ownerId).Find(&groupList)
            
            // 3. 构建响应数据
            var groupListRsp []respond.LoadMyGroupRespond
            for _, group := range groupList {
                groupListRsp = append(groupListRsp, ...)
            }
            
            // 4. 序列化并写入缓存
            rspString, _ := json.Marshal(groupListRsp)
            myredis.SetKeyEx("contact_mygroup_list_"+ownerId, string(rspString), time.Minute*constants.REDIS_TIMEOUT)
            
            return "获取成功", groupListRsp, 0
        }
        return constants.SYSTEM_ERROR, nil, -1
    }
    
    // 5. 缓存存在，直接反序列化返回
    var groupListRsp []respond.LoadMyGroupRespond
    json.Unmarshal([]byte(rspString), &groupListRsp)
    return "获取成功", groupListRsp, 0
}
```

### 8.5 缓存删除时机

**创建群聊后**：删除用户群聊列表缓存
```go
// CreateGroup 创建群聊
func (g *groupInfoService) CreateGroup(groupReq request.CreateGroupRequest) (string, int) {
	// ... 创建群聊逻辑

	// ★删除缓存，下次查询会重新从数据库读取最新数据（带错误处理）
	if err := myredis.DelKeysWithPattern("contact_mygroup_list_" + groupReq.OwnerId); err != nil {
		zlog.Error(err.Error())
	}
	return "创建成功", 0
}
```

**进群后**：删除用户相关缓存

## 十、总结
### 10.1 核心要点回顾
| 接口 | 核心要点 |
|------|------|
| createGroup | Members需要json.Marshal序列化，初始只有群主，需要创建UserContact |
| loadMyGroup | 查询owner_id字段，按created_at倒序排列，返回简化的DTO |
| checkGroupAddMode | 只返回AddMode数值，用于判断加群流程 |
| getGroupInfo | 返回详细DTO，包含IsDeleted判断软删除状态 |
### 10.2 最佳实践
1. **JSON序列化**: 对于复杂数据类型（数组、切片），使用json.Marshal/Unmarshal处理
2. **DTO设计**: 只返回前端需要的字段，减少数据传输
3. **软删除**: 使用DeletedAt.Valid判断删除状态，不返回敏感字段
4. **错误处理**: 所有数据库操作都需要检查Error，记录日志
5. **唯一标识**: 群组Uuid以"G"开头，11位随机字符，保证唯一性
### 10.3 常见问题
| 问题 | 解决方案 |
|------|------|
| Members字段如何读取? | 使用json.Unmarshal反序列化 |
| 群组Uuid如何生成? | "G" + 11位随机字符串 |
| 为什么初始MemberCnt=1? | 创建时只有群主一人 |
| 如何判断群聊已解散? | 检查DeletedAt.Valid字段 |
