### Controller函数模板

```go
// 标准Controller函数模板
func XxxController(c *gin.Context) {
    // 1. 绑定请求参数
    var req request.XxxRequest
    if err := c.BindJSON(&req); err != nil {
        zlog.Error(err.Error())
        c.JSON(http.StatusOK, gin.H{
            "code":    500,
            "message": constants.SYSTEM_ERROR,
        })
        return
    }
    
    // 2. 调用Service层
    message, data, ret := gorm.XxxService.XxxMethod(req)
    
    // 3. 返回响应
    https_server.JsonBack(c, message, ret, data)
}
```

```go
type xxxService struct{}

// Service方法模板
func (x *xxxService) XxxMethod(req request.XxxRequest) (string, *respond.XxxRespond, int) {
    // 1. 数据库查询/操作
    var data model.Xxx
    if res := dao.GormDB.First(&data, "uuid = ?", req.Uuid); res.Error != nil {
        zlog.Error(res.Error.Error())

    }
    
    // 2. 构建响应DTO
    rsp := &respond.XxxRespond{
        Uuid: data.Uuid,
        // ...其他字段
    }
    
    return "操作成功", rsp, 0
```

### 1. 软删除数据如何查询？

**问题：** 如何查询已被软删除的数据？

**解决方案：**
```go

// SQL: SELECT * FROM user_info WHERE deleted_at IS NULL

// Unscoped查询：返回所有数据，包括软删除的
dao.GormDB.Unscoped().Find(&users)

// 判断是否软删除

    // 已软删除

    // 正常数据
```

### 2. JSON字段如何处理？

**问题：** GroupInfo的Members字段是JSON类型，如何读取和修改？

**解决方案：**
```go
// 数据库模型定义
type GroupInfo struct {
    Members json.RawMessage `gorm:"column:members;type:json"`
}

// ===== 读取JSON字段 =====
var group model.GroupInfo
dao.GormDB.First(&group, "uuid = ?", groupId)

// 反序列化：JSON -> Go切片
var members []string
if err := json.Unmarshal(group.Members, &members); err != nil {
    zlog.Error(err.Error())
}
// 结果：members = ["U001", "U002", "U003"]

// ===== 添加成员 =====
members = append(members, "U004")  // 添加新成员

// 序列化：Go切片 -> JSON
if data, err := json.Marshal(members); err != nil {
    zlog.Error(err.Error())
} else {
    group.Members = data
}

// ===== 删除成员 =====
for i, member := range members {
    if member == "U002" {
        members = append(members[:i], members[i+1:]...)
        break
    }
}

// 保存更新
dao.GormDB.Save(&group)
```

**注意事项：**
- `json.RawMessage`底层是`[]byte`类型
- 每次操作都需要序列化/反序列化
- 操作完成后必须保存到数据库

### 3. 批量操作如何实现？

**问题：** 如何批量更新或删除多条记录？

**解决方案：**
```go
// ===== 批量查询 =====
// 使用IN条件
var users []model.UserInfo
dao.GormDB.Where("uuid IN ?", uuidList).Find(&users)
// SQL: SELECT * FROM user_info WHERE uuid IN ('U001', 'U002', 'U003')

// ===== 批量更新 =====
// 方式1：使用Model + Where + Update
dao.GormDB.Model(&model.UserInfo{}).
    Where("uuid IN ?", uuidList).
    Update("status", 1)
// SQL: UPDATE user_info SET status = 1 WHERE uuid IN ('U001', 'U002')

// 方式2：使用Updates批量更新多个字段
dao.GormDB.Model(&model.UserInfo{}).
    Where("uuid IN ?", uuidList).
    Updates(map[string]interface{}{
        "status": 1,
        "is_admin": 0,
    })

// ===== 批量删除（软删除） =====
var deletedAt gorm.DeletedAt
deletedAt.Time = time.Now()
deletedAt.Valid = true

dao.GormDB.Model(&model.UserInfo{}).
    Where("uuid IN ?", uuidList).
    Update("deleted_at", deletedAt)
// SQL: UPDATE user_info SET deleted_at = '2024-01-15' WHERE uuid IN (...)
```

**性能建议：**
- 批量操作优于循环单条操作
- 大批量操作考虑使用事务
- 注意批量操作的条件范围

### 5. Redis缓存如何使用？

**问题：** 如何使用Redis缓存用户数据？

**解决方案：**
```go
// ===== 存储数据 =====
key := "user_info_" + uuid
value := json.Marshal(userInfo)
redis.SetKey(key, value, 3600)  // 缓存1小时

// ===== 读取数据 =====
data, err := redis.GetKey(key)
if err != nil {
    // 缓存不存在，从数据库查询
    dao.GormDB.First(&user, "uuid = ?", uuid)
    // 存入缓存
    redis.SetKey(key, json.Marshal(user), 3600)
} else {
    // 从缓存解析
    json.Unmarshal(data, &user)
}

// ===== 删除缓存 =====
// 单个删除
redis.DelKeyIfExists(key)

// 批量删除（使用模式匹配）
redis.DelKeysWithPrefix("user_info_")  // 删除所有user_info_开头的key
```

```go
    dao.GormDB.Model(&model.UserContact{}).

            "status": 4,

        })
    
    // 3. 删除双方的会话记录
    // ...
}
```
