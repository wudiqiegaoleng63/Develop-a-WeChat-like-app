### store
```javascript
import { createStore } from 'vuex'

export default createStore({
  state: {
    // web服务器地址
    // backendUrl: 'https://192.168.2.114:8000',
    // wsUrl: 'wss://192.168.2.114:8000',
    backendUrl: 'https://127.0.0.1:8000',
    wsUrl: 'wss://127.0.0.1:8000',
    userInfo: (sessionStorage.getItem('userInfo') && JSON.parse(sessionStorage.getItem('userInfo'))) || {},
    socket: null,
  },
  getters: {
  },
  mutations: {
    setUserInfo(state, userInfo) {
      state.userInfo = userInfo;
      sessionStorage.setItem('userInfo', JSON.stringify(userInfo));
    },
    cleanUserInfo(state) {
      state.userInfo = {};
      sessionStorage.removeItem('userInfo');
    }
  },
  actions: {
  },
  modules: {
  }
})
```

前端项目使用store作为状态管理模块，存储一些全局使用的变量，比如说backendUrl就是后端https服务器的ip端口地址，wsUrl就是后端websocket连接的ip端口地址，以及userInfo存储当前前端的用户信息，比如昵称、头像、性别、电话、邮箱、生日等（这里使用sessionStorage，在浏览器中按照会话为单位来存储userInfo，在浏览器中按F12打开开发者工具，点击应用程序，可以看到此时的会话存储），

### router
<font style="color:rgba(0, 0, 0, 0.75);">Vue Router是一个官方的Vue.js路由管理器，它与 Vue.js 核心深度集成，通过它可以轻松地为单页应用程序（SPA）提供路由管理和导航功能。</font>

> <font style="color:rgba(0, 0, 0, 0.75);">Vue Router有一些功能。</font>
>
> <font style="color:rgba(0, 0, 0, 0.75);">路由器：Vue Router 提供了一个路由器，用于管理应用程序中的路由。Vue Router 实例化一个 Vue Router 对象，注册路由规则，并以它为中心连接其他组件。</font>
>
> <font style="color:rgba(0, 0, 0, 0.75);"></font>
>
> <font style="color:rgba(0, 0, 0, 0.75);">路由：路由是分发到不同组件的 URL 地址。在 Vue Router 中，路由通常是由 path 规则和相应的组件定义的。当浏览器的 URL 匹配到路由的 path 后，相应的组件将会被加载到页面中。路由的信息可以从 route 对象中获取。</font>
>
> <font style="color:rgba(0, 0, 0, 0.75);"></font>
>
> <font style="color:rgba(0, 0, 0, 0.75);">路由规则：路由规则是由 path、component、name、meta、props 等属性组成的。其中，path 表示 URL 的路径，component 表示要渲染的组件，name 表示路由名称，meta 表示路由的元数据，props 表示路由 props 数据。路由规则可以注册到 Vue Router 中。</font>
>
> <font style="color:rgba(0, 0, 0, 0.75);"></font>
>
> <font style="color:rgba(0, 0, 0, 0.75);">导航守卫：导航守卫是在路由跳转时执行的钩子函数，用于控制路由的访问权限、处理路由跳转前后的逻辑等。在 Vue Router 中，对于选项式 API，常用的导航守卫有 beforeRouteEnter、beforeRouteUpdate、beforeRouteLeave 等；对于使用组合 API 和 setup 函数来编写组件的，可以通过 onBeforeRouteUpdate 和 onBeforeRouteLeave 分别添加 update 和 leave 守卫。</font>
>

```javascript
import { createRouter, createWebHistory } from 'vue-router'
import store from '../store/index.js'

const routes = [
  {
    path: '/',
    redirect: { name: 'Login' }
  },
  {
    path: '/login',
    name: 'Login',
    component: () => import('../views/access/Login.vue')
  },
  {
    path: '/smsLogin',
    name: 'smsLogin',
    component: () => import('../views/access/SmsLogin.vue')
  },
  {
    path: '/register',
    name: 'Register',
    component: () => import('../views/access/Register.vue')
  },
  {
    path: '/chat/owninfo',
    name: 'OwnInfo',
    component: () => import('../views/chat/user/OwnInfo.vue')
  },
  {
    path: '/chat/contactlist',
    name: 'ContactList',
    component: () => import('../views/chat/contact/ContactList.vue')
  },
  {
    path: '/chat/:id',
    name: 'ContactChat',
    component: () => import('../views/chat/contact/ContactChat.vue')
  },
  {
    path: '/chat/sessionList',
    name: 'SessionList',
    component: () => import('../views/chat/session/SessionList.vue')
  },
  {
    path: '/manager',
    name: 'Manager',
    component: () => import('../views/manager/Manager.vue')
  }
]

const router = createRouter({
  history: createWebHistory(),
  routes
});

router.beforeEach((to, from, next) => {
  if (!store.state.userInfo.uuid) {
    if (to.path === '/login' || to.path === '/register' || to.path === '/smsLogin') {
      next()
      return
    }
    next('/login')
  } else {
    next()
  }
})

export default router
```

这里创建了所有router链路。并且在每次跳转下一个页面时，都事先判断该会话是否已经登录了，即是否存在用户信息。如果不存在，得判断下一跳的页面是否是登录页面、注册页面或者验证码登录页面，如果是就直接跳转，否则都会先跳转到登录页面进行登录，并让会话存入用户信息。如果用户信息存在的话，也是直接跳转就可以了。

当某一个页面要跳转到另一个页面时，只需要引入useRouter，并且在js函数中使用router.push('/xxxx')即可完成跳转。

更详细的对router的介绍可以看[一文详解:Vue3中使用Vue Router_vue3 router-CSDN博客](https://blog.csdn.net/w137160164/article/details/131152766?ops_request_misc=%257B%2522request%255Fid%2522%253A%25227bea6d754a62b33489d03a9460bbedb1%2522%252C%2522scm%2522%253A%252220140713.130102334..%2522%257D&request_id=7bea6d754a62b33489d03a9460bbedb1&biz_id=0&utm_medium=distribute.pc_search_result.none-task-blog-2~all~top_click~default-4-131152766-null-null.142^v101^pc_search_result_base3&utm_term=vue%20router&spm=1018.2226.3001.4187)

const handleLogin = async () => {
  try {
    if (!data.loginData.telephone || !data.loginData.password) {
      ElMessage.error("请填写完整登录信息。");
      return;
    }
    if (!checkTelephoneValid()) {
      ElMessage.error("请输入有效的手机号码。");
      return;
    }
    const response = await axios.post(
      store.state.backendUrl + "/login",
      data.loginData

    if (response.data.code == 200) {
      if (response.data.data.status == 1) {
        ElMessage.error("该账号已被封禁，请联系管理员。");
        return;
      }
      try {
        ElMessage.success(response.data.message);

          response.data.data.avatar =
            store.state.backendUrl + response.data.data.avatar;
        }
        store.commit("setUserInfo", response.data.data);
        
        // 创建websocket连接
        
        router.push("/chat/sessionlist");
      } catch (error) {
        console.log(error);
      }
    } else {
      ElMessage.error(response.data.message);
    }
  } catch (error) {
    ElMessage.error(error);
  }

<font style="color:rgb(6, 7, 31);background-color:rgb(253, 253, 254);">使用从响应中接收到的数据（</font>`<font style="color:rgb(6, 7, 31);">rsp.data</font>`<font style="color:rgb(6, 7, 31);background-color:rgb(253, 253, 254);">）创建一个新的 Blob 对象。Blob 对象表示一个不可变的原始数据的类文件对象。这里还设置了 Blob 的 MIME 类型，如果响应头中没有提供 </font>`<font style="color:rgb(6, 7, 31);">content-type</font>`<font style="color:rgb(6, 7, 31);background-color:rgb(253, 253, 254);">，则默认为 </font>`<font style="color:rgb(6, 7, 31);">"application/octet-stream"</font>`<font style="color:rgb(6, 7, 31);">。</font>

当offer，answer都发放完毕之后，发起端会开始查询本端的后选择，将所有的候选者消息收集之后，会触发onIceCandidate方法。该方法会创建一个PROXY代理消息，类型为candidate。通过前端的websocket连接将候选者信息转发给后端。后端发送给对端用户。

```vue
const handleCandidate = (val) => {
  data.rtcPeerConn.addIceCandidate(new RTCIceCandidate(val));
};
```

当对端用户接收到本端的候选者信息会添加到远端连接中。



到此音视频通话连接成功，p2p连接完全建立。
