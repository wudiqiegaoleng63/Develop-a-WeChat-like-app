## 一、整体流程回顾

### 开发顺序（重要）

```
Model → DAO → Service → Controller → 路由注册

① Model层: 用户模型已在前序文档定义（user_info.go）
② DAO层: 数据库连接已在前序文档完成

④ Controller层: HTTP请求处理（user_info_controller.go）

```

```

         -> Service(处理结果) -> Controller(JsonBack返回) -> 前端响应
```

```go
// UpdateUserInfo - 更新用户信息接口

func UpdateUserInfo(c *gin.Context) {
    // 1. 绑定请求参数
    var req request.UpdateUserInfoRequest
    // ★c.BindJSON: 将JSON请求体绑定到结构体
    if err := c.BindJSON(&req); err != nil {
```

### 5. Service层完整代码

**文件位置:** `internal/service/gorm/user_info_service.go`

```go
// ============================================================
// UpdateUserInfo - 更新用户信息
// ============================================================
// ★注释说明：某用户修改了信息，可能会影响contact_user_list，不需要删除redis的contact_user_list，
//           timeout之后会自己更新。但是需要更新redis的user_info，因为可能影响用户搜索
func (u *userInfoService) UpdateUserInfo(updateReq request.UpdateUserInfoRequest) (string, int) {
    // 1. 根据uuid查询用户
    var user model.UserInfo
    // ★dao.GormDB.First: 查询第一条匹配记录
    if res := dao.GormDB.First(&user, "uuid = ?", updateReq.Uuid); res.Error != nil {
        zlog.Error(res.Error.Error())
        return constants.SYSTEM_ERROR, -1
    }
    
    // 2. 按需更新字段（只更新请求中提供的非空字段）
    // ★选择性更新：避免覆盖用户未修改的字段
    if updateReq.Email != "" {
        user.Email = updateReq.Email
    }
    if updateReq.Nickname != "" {
        user.Nickname = updateReq.Nickname
    }
    if updateReq.Birthday != "" {
        user.Birthday = updateReq.Birthday
    }
    if updateReq.Signature != "" {
        user.Signature = updateReq.Signature
    }
    if updateReq.Avatar != "" {
        user.Avatar = updateReq.Avatar
    }
    
    // 3. 保存更新到数据库
    // ★dao.GormDB.Save: 保存所有字段，更新数据库记录
    if res := dao.GormDB.Save(&user); res.Error != nil {
        zlog.Error(res.Error.Error())
        return constants.SYSTEM_ERROR, -1
    }
    
    // ★4. 删除Redis缓存（当前代码被注释，暂不启用）
    // 更新用户信息后，需要删除user_info缓存，下次查询会从数据库重新加载最新数据
    //if err := myredis.DelKeysWithPattern("user_info_" + updateReq.Uuid); err != nil {
    //    zlog.Error(err.Error())
    //}
    
    return "修改用户信息成功", 0
}
```

### 6. 数据库操作说明

```go
// 查询用户
dao.GormDB.First(&user, "uuid = ?", updateReq.Uuid)
// SQL: SELECT * FROM user_info WHERE uuid = 'U12345678901' LIMIT 1

// 更新用户
dao.GormDB.Save(&user)
// SQL: UPDATE user_info SET nickname='新昵称', email='new@email.com', ... WHERE uuid = 'U12345678901'

// Redis缓存Key格式（被注释状态）
// "user_info_" + uuid  例如: "user_info_U12345678901"
```

### 7. 关键概念详解

#### 选择性更新策略

```go
// ★只更新请求中提供的非空字段
if updateReq.Nickname != "" {
    user.Nickname = updateReq.Nickname
}
// 这样可以避免将空字符串写入数据库，覆盖原有数据
```

#### Redis缓存清理说明

```go
// ★源码注释说明需要删除user_info缓存：
// "某用户修改了信息...需要更新redis的user_info，因为可能影响用户搜索"
// 但当前代码被注释，暂不启用缓存清理

// 启用后的逻辑：更新DB → 删除缓存 → 下次查询从DB重新加载
//if err := myredis.DelKeysWithPattern("user_info_" + updateReq.Uuid); err != nil {
//    zlog.Error(err.Error())
//}
```

#### Save vs Update

```go
// Save: 保存整个结构体，会更新所有字段
dao.GormDB.Save(&user)

// Update: 只更新指定字段
dao.GormDB.Model(&user).Update("nickname", "新昵称")
```

```go
// ============================================================
// GetUserInfo - 获取用户信息
// ============================================================
func (u *userInfoService) GetUserInfo(uuid string) (string, *respond.GetUserInfoRespond, int) {
	zlog.Info(uuid)

	rspString, err := myredis.GetKeyNilIsErr("user_info_" + uuid)

	if err != nil {
		// 缓存未命中
		if errors.Is(err, redis.Nil) {
			zlog.Info(err.Error())

			// 查询用户
			var user model.UserInfo
			if res := dao.GormDB.Where("uuid=?", uuid).Find(&user); res.Error != nil {
				zlog.Error(res.Error.Error())
				return constants.SYSTEM_ERROR, nil, -1
			}

			rsp := respond.GetUserInfoRespond{
				Uuid:       user.Uuid,
				Telephone:  user.Telephone,
				Nickname:   user.Nickname,
				Avatar:     user.Avatar,
				Birthday:   user.Birthday,
				Email:      user.Email,
				Gender:     user.Gender,
				Signature:  user.Signature,
				CreatedAt:  user.CreatedAt.Format("2006-01-02 15:04:05"),
				IsAdmin:    user.IsAdmin,
				Status:     user.Status,
			}

			// 写入缓存
			rspBytes, err := json.Marshal(rsp)
			if err != nil {
				zlog.Error(err.Error())
			}

			if err := myredis.SetKeyEx("user_info_"+uuid, string(rspBytes), constants.REDIS_TIMEOUT*time.Minute); err != nil {
				zlog.Error(err.Error())
			}
			return "获取用户信息成功", &rsp, 0
		} else {
			// 其他Redis错误
			zlog.Error(err.Error())
			return constants.SYSTEM_ERROR, nil, -1
		}
	}

	// 缓存命中
	var rsp respond.GetUserInfoRespond
	if err := json.Unmarshal([]byte(rspString), &rsp); err != nil {
		zlog.Error(err.Error())
	}
	return "获取用户信息成功", &rsp, 0
}
```

### 6. 数据库操作说明

```go
// 条件查询
dao.GormDB.Where("uuid = ?", uuid).Find(&user)
// SQL: SELECT * FROM user_info WHERE uuid = 'U12345678901'

// Redis缓存Key格式
// "user_info_" + uuid  例如: "user_info_U12345678901"
```

### 7. 关键概念详解

#### Redis缓存策略（读缓存模式）

```go
// ★当前实现的缓存策略：
// 1. 先查询Redis缓存（myredis.GetKeyNilIsErr）
// 2. 缓存命中 → 直接解析JSON返回
// 3. 缓存未命中 → 查询数据库 → 构建响应 → （写入缓存暂被注释）

// Redis.Nil错误判断
if errors.Is(err, redis.Nil) {
    // key不存在，需要从数据库查询
}
```

#### DTO（数据传输对象）

DTO是数据传输对象（Data Transfer Object），用于在网络层传输数据。它的作用：

1. **减少调用次数**: 一次性传输所有需要的数据
2. **数据格式转换**: 将数据库模型转换为前端需要的格式
3. **隐藏敏感信息**: 不暴露password等敏感字段

#### 时间格式化

```go
// ★Go的时间格式化使用固定模板"2006-01-02 15:04:05"
user.CreatedAt.Format("2006-01-02 15:04:05")
// 输出: "2024-01-15 10:30:45"
```

```go
// ============================================================
// GetUserInfoList - 获取用户列表（管理员）
// ============================================================
// ★管理员功能不使用Redis缓存，因为用户数据变化频繁
func (u *userInfoService) GetUserInfoList(ownerId string) (string, []respond.GetUserListRespond, int) {
    // 1. 从数据库获取所有用户（排除管理员自己）
    var users []model.UserInfo
    // ★Unscoped(): 查询包含软删除的数据，管理员需要看到所有用户
    if res := dao.GormDB.Unscoped().Where("uuid != ?", ownerId).Find(&users); res.Error != nil {
        zlog.Error(res.Error.Error())
        return constants.SYSTEM_ERROR, nil, -1
    }
    
    // 2. 构建响应列表
    var rsp []respond.GetUserListRespond
    for _, user := range users {
        rp := respond.GetUserListRespond{
            Uuid:      user.Uuid,
            Telephone: user.Telephone,
            Nickname:  user.Nickname,
            Status:    user.Status,
            IsAdmin:   user.IsAdmin,
        }
        // ★检查软删除状态
        if user.DeletedAt.Valid {
            rp.IsDeleted = true
        } else {
            rp.IsDeleted = false
        }
        rsp = append(rsp, rp)
    }
    
    return "获取用户列表成功", rsp, 0
}
```

### 6. 数据库操作说明

```go
// Unscoped查询（包含软删除数据）
dao.GormDB.Unscoped().Where("uuid != ?", ownerId).Find(&users)
// SQL: SELECT * FROM user_info WHERE uuid != 'U12345678901'
// ★正常查询会自动添加 WHERE deleted_at IS NULL 条件
// ★Unscoped() 会忽略软删除条件，查询所有数据
```

### 7. 关键概念详解

#### Unscoped()的作用

```go
// 正常查询: 只返回未删除的数据
dao.GormDB.Find(&users)
// SQL: SELECT * FROM user_info WHERE deleted_at IS NULL

// Unscoped查询: 返回所有数据，包括软删除的
dao.GormDB.Unscoped().Find(&users)
// SQL: SELECT * FROM user_info
```

#### 为什么管理员功能不用Redis缓存

```go
// ★管理员功能暂不使用Redis缓存的原因：
// 1. 管理员数量少，缓存利用率低
// 2. 用户数据变化频繁（启用/禁用/删除等操作）
// 3. 频繁更新Redis反而增加系统开销
```

```go
// ============================================================
// AbleUsers - 启用用户
// ============================================================
// ★注释说明：用户启用禁用状态变化会影响contact_user_list状态，
//           需要实时更新，所以Redis缓存清理代码暂被注释
func (u *userInfoService) AbleUsers(uuidList []string) (string, int) {
    // 1. 批量查询用户
    var users []model.UserInfo
    // ★Where "uuid in (?)": 批量条件查询

        zlog.Error(res.Error.Error())
```

```go
// ============================================================
// DeleteUsers - 删除用户（软删除）
// ============================================================
// ★注释说明：用户是否启用禁用需要实时更新contact_user_list状态，
//           所以Redis的contact_user_list需要删除
func (u *userInfoService) DeleteUsers(uuidList []string) (string, int) {
    // 1. 批量查询用户
    var users []model.UserInfo
    if res := dao.GormDB.Model(model.UserInfo{}).Where("uuid in (?)", uuidList).Find(&users); res.Error != nil {
        zlog.Error(res.Error.Error())
        return constants.SYSTEM_ERROR, -1
    }
    
    // 2. 处理每个用户
    for _, user := range users {
        // 2.1 软删除用户
        user.DeletedAt.Valid = true
        user.DeletedAt.Time = time.Now()
        if res := dao.GormDB.Save(&user); res.Error != nil {
            zlog.Error(res.Error.Error())
            return constants.SYSTEM_ERROR, -1
        }
        
        // 2.2 软删除相关会话
        var sessionList []model.Session
        if res := dao.GormDB.Where("send_id = ? or receive_id = ?", user.Uuid, user.Uuid).Find(&sessionList); res.Error != nil {
            if errors.Is(res.Error, gorm.ErrRecordNotFound) {
                zlog.Info(res.Error.Error())  // 没有会话记录，继续执行
            } else {
                zlog.Error(res.Error.Error())
                return constants.SYSTEM_ERROR, -1
            }
        }
        for _, session := range sessionList {
            var deletedAt gorm.DeletedAt
            deletedAt.Time = time.Now()
            deletedAt.Valid = true
            session.DeletedAt = deletedAt
            if res := dao.GormDB.Save(&session); res.Error != nil {
                zlog.Error(res.Error.Error())
                return constants.SYSTEM_ERROR, -1
            }
        }
        
        // 2.3 软删除相关联系人
        var contactList []model.UserContact
        if res := dao.GormDB.Where("user_id = ? or contact_id = ?", user.Uuid, user.Uuid).Find(&contactList); res.Error != nil {
            if errors.Is(res.Error, gorm.ErrRecordNotFound) {
                zlog.Info(res.Error.Error())
            } else {
                zlog.Error(res.Error.Error())
                return constants.SYSTEM_ERROR, -1
            }
        }
        for _, contact := range contactList {
            var deletedAt gorm.DeletedAt
            deletedAt.Time = time.Now()
            deletedAt.Valid = true
            contact.DeletedAt = deletedAt
            if res := dao.GormDB.Save(&contact); res.Error != nil {
                zlog.Error(res.Error.Error())
                return constants.SYSTEM_ERROR, -1
            }
        }
        
        // 2.4 软删除相关申请记录
        var applyList []model.ContactApply
        if res := dao.GormDB.Where("user_id = ? or contact_id = ?", user.Uuid, user.Uuid).Find(&applyList); res.Error != nil {
            if errors.Is(res.Error, gorm.ErrRecordNotFound) {
                zlog.Info(res.Error.Error())
            } else {
                zlog.Error(res.Error.Error())
                return constants.SYSTEM_ERROR, -1
            }
        }
        for _, apply := range applyList {
            var deletedAt gorm.DeletedAt
            deletedAt.Time = time.Now()
            deletedAt.Valid = true
            apply.DeletedAt = deletedAt
            if res := dao.GormDB.Save(&apply); res.Error != nil {
                zlog.Error(res.Error.Error())
                return constants.SYSTEM_ERROR, -1
            }
        }
    }
    
    // ★3. 清除Redis缓存（当前代码被注释，暂不启用）
    // 删除所有"contact_user_list"开头的key
    //if err := myredis.DelKeysWithPrefix("contact_user_list"); err != nil {
    //    zlog.Error(err.Error())
    //}
    
    return "删除用户成功", 0
}
```

```go
// 1. user_info      - 用户基本信息

// 3. user_contact   - 用户联系人关系
// 4. contact_apply  - 用户申请记录
```

### 7. 关键概念详解

#### 软删除的级联效应

```
删除用户 -> 软删除UserInfo -> 软删除Session -> 软删除UserContact -> 软删除ContactApply -> （Redis缓存清理暂被注释）
```

★删除用户时，需要同时删除与之相关的所有数据：
- Session表：send_id或receive_id包含该用户的所有会话
- UserContact表：user_id或contact_id包含该用户的联系人关系
- ContactApply表：user_id或contact_id包含该用户的申请记录

#### 软删除的优势

```go
// ★软删除 vs 硬删除
// 硬删除: DELETE FROM user_info WHERE uuid = 'U12345678901'
//        数据完全消失，无法恢复

// 软删除: UPDATE user_info SET deleted_at = '2024-01-15' WHERE uuid = 'U12345678901'
//        数据保留，只是标记为已删除
//        优势：可以恢复数据、维护索引开销更低、保留审计记录
```
